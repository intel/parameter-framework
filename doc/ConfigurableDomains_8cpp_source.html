68 void CConfigurableDomains::apply(CParameterBlackboard* pParameterBlackboard,
69  CSyncerSet& syncerSet,
70  bool bForce,
71  core::Results& infos) const
72 {
74 
75  // Start with domains that can be synchronized all at once (with passed syncer set)
76  size_t uiNbConfigurableDomains = getNbChildren();
77 
78  for (size_t child = 0; child < uiNbConfigurableDomains; child++) {
79 
80  const CConfigurableDomain* pChildConfigurableDomain = static_cast<const CConfigurableDomain*>(getChild(child));
81 
82  std::string info;
83  // Apply and collect syncers when relevant
84  pChildConfigurableDomain->apply(pParameterBlackboard, &syncerSet, bForce, info);
85 
86  if (!info.empty()) {
87  infos.push_back(info);
88  }
89  }
90  // Synchronize those collected syncers
91  syncerSet.sync(*pParameterBlackboard, false, NULL);
92 
93  // Then deal with domains that need to synchronize along apply
94  for (size_t child = 0; child < uiNbConfigurableDomains; child++) {
95 
96  const CConfigurableDomain* pChildConfigurableDomain = static_cast<const CConfigurableDomain*>(getChild(child));
97 
98  std::string info;
99  // Apply and synchronize when relevant
100  pChildConfigurableDomain->apply(pParameterBlackboard, NULL, bForce, info);
101  if (!info.empty()) {
102  infos.push_back(info);
103  }
104  }
105 }
