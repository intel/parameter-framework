176 // Generic encodability check
177 template <typename type>
178 bool CParameterType::doIsEncodable(type data, bool bIsSigned) const
179 {
180  if (getSize() == sizeof(data)) {
181  // Prevent inappropriate shifts
182  return true;
183  }
184 
185  uint32_t uiShift = getSize() * 8;
186 
187  if (!bIsSigned) {
188 
189  // Check high bits are clean
190  return !(data >> uiShift);
191 
192  } else {
193 
194  // Negative value?
195  bool bIsValueExpectedNegative = (data & (1 << (uiShift - 1))) != 0;
196 
197  // Check high bits are clean
198  return bIsValueExpectedNegative ? !(~data >> uiShift) : !(data >> uiShift);
199  }
200 }

202 // Remove all bits set outside available range
203 uint32_t CParameterType::makeEncodable(uint32_t uiData) const
204 {
205  if (getSize() == sizeof(uint32_t)) {
206 
207  return uiData;
208  }
209  uint32_t uiSizeInBits = getSize() * 8;
210 
211  uint32_t uiMask = (1 << uiSizeInBits) - 1;
212 
213  return uiData & uiMask;
214 }
