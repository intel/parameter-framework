431 bool CTestPlatform::setCriterionStateByLexicalSpace(const IRemoteCommand& remoteCommand,
432  string& strResult)
433 {
434 
435  // Get criterion name
436  std::string strCriterionName = remoteCommand.getArgument(0);
437 
438  ISelectionCriterionInterface* pCriterion =
439  _pParameterMgrPlatformConnector->getSelectionCriterion(strCriterionName);
440 
441  if (!pCriterion) {
442 
443  strResult = "Unable to retrieve selection criterion: " + strCriterionName;
444 
445  return false;
446  }
447 
448  // Get criterion type
449  const ISelectionCriterionTypeInterface* pCriterionType = pCriterion->getCriterionType();
450 
451  // Get substate number, the first argument (index 0) is the criterion name
452  uint32_t uiNbSubStates = remoteCommand.getArgumentCount() - 1;
453 
454  // Check that exclusive criterion has only one substate
455  if (!pCriterionType->isTypeInclusive() && uiNbSubStates != 1) {
456 
457  strResult = "Exclusive criterion " + strCriterionName + " can only have one state";
458 
459  return false;
460  }
461 
463  int iNumericalState = 0;
464  uint32_t uiLexicalSubStateIndex;
465 
466  // Parse lexical substates
467  std::string strLexicalState = "";
468 
469  for (uiLexicalSubStateIndex = 1;
470  uiLexicalSubStateIndex <= uiNbSubStates;
471  uiLexicalSubStateIndex++) {
472  /*
473  * getNumericalValue method from ISelectionCriterionTypeInterface strip his parameter
474  * first parameter based on | sign. In case that the user uses multiple parameters
475  * to set InclusiveCriterion value, we aggregate all desired values to be sure
476  * they will be handled correctly.
477  */
478  if (uiLexicalSubStateIndex != 1) {
479  strLexicalState += "|";
480  }
481  strLexicalState += remoteCommand.getArgument(uiLexicalSubStateIndex);
482  }
483 
484  // Translate lexical to numerical substate
485  if (!pCriterionType->getNumericalValue(strLexicalState, iNumericalState)) {
486 
487  strResult = "Unable to find lexical state \""
488  + strLexicalState + "\" in criteria " + strCriterionName;
489 
490  return false;
491  }
492 
493  // Set criterion new state
494  pCriterion->setCriterionState(iNumericalState);
495 
496  return true;
497 }
