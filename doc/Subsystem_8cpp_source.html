406 // Subsystem object creation handling
407 bool CSubsystem::handleSubsystemObjectCreation(
408  CInstanceConfigurableElement* pInstanceConfigurableElement,
409  CMappingContext& context, bool& bHasCreatedSubsystemObject, string& strError)
410 {
411  uint32_t uiItem;
412  bHasCreatedSubsystemObject = false;
413 
414  for (uiItem = 0; uiItem < _subsystemObjectCreatorArray.size(); uiItem++) {
415 
416  const CSubsystemObjectCreator* pSubsystemObjectCreator =
417  _subsystemObjectCreatorArray[uiItem];
418 
419  // Mapping key
420  string strKey = pSubsystemObjectCreator->getMappingKey();
421  // Object id
422  const string* pStrValue;
423 
424  if (pInstanceConfigurableElement->getMappingData(strKey, pStrValue)) {
425 
426  // First check context consistency
427  // (required ancestors must have been set prior to object creation)
428  uint32_t uiAncestorKey;
429  uint32_t uiAncestorMask = pSubsystemObjectCreator->getAncestorMask();
430 
431  for (uiAncestorKey = 0; uiAncestorKey < _contextMappingKeyArray.size(); uiAncestorKey++) {
432 
433  if (!((1 << uiAncestorKey) & uiAncestorMask)) {
434  // Ancestor not required
435  continue;
436  }
437  // Check ancestor was provided
438  if (!context.iSet(uiAncestorKey)) {
439 
440  strError = getMappingError(strKey, _contextMappingKeyArray[uiAncestorKey] +
441  " not set", pInstanceConfigurableElement);
442 
443  return false;
444  }
445  }
446 
447  // Then check configurable element size is correct
448  if (pInstanceConfigurableElement->getFootPrint() >
449  pSubsystemObjectCreator->getMaxConfigurableElementSize()) {
450 
451  string strSizeError = "Size should not exceed " +
452  CUtility::toString(
453  pSubsystemObjectCreator->getMaxConfigurableElementSize());
454 
455  strError = getMappingError(strKey, strSizeError, pInstanceConfigurableElement);
456 
457  return false;
458  }
459 
460  // Do create object and keep its track
461  _subsystemObjectList.push_back(pSubsystemObjectCreator->objectCreate(
462  *pStrValue, pInstanceConfigurableElement, context));
463 
464  // Indicate subsytem creation to caller
465  bHasCreatedSubsystemObject = true;
466 
467  // The subsystem Object has been instantiated, no need to continue looking for an
468  // instantiation mapping
469  break;
470  }
471  }
472 
473  return true;
474 }

476 // From IMapper
477 // Handle a configurable element mapping
478 bool CSubsystem::mapBegin(CInstanceConfigurableElement* pInstanceConfigurableElement,
479  bool& bKeepDiving, string& strError)
480 {
481  // Get current context
482  CMappingContext context = _contextStack.top();
483 
484  // Add mapping in context
485  if (!handleMappingContext(pInstanceConfigurableElement, context,
486  strError)) {
487 
488  return false;
489  }
490 
491  // Push context
492  _contextStack.push(context);
493 
494  // Assume diving by default
495  bKeepDiving = true;
496 
497  // Deal with ambiguous usage of parameter blocks
498  bool bShouldCreateSubsystemObject = true;
499 
500  switch(pInstanceConfigurableElement->getType()) {
501 
502  case CInstanceConfigurableElement::EComponent:
503  case CInstanceConfigurableElement::EParameterBlock:
504  // Subsystem object creation is optional in parameter blocks
505  bShouldCreateSubsystemObject = false;
506  // No break
507  case CInstanceConfigurableElement::EBitParameterBlock:
508  case CInstanceConfigurableElement::EParameter:
509  case CInstanceConfigurableElement::EStringParameter:
510 
511  bool bHasCreatedSubsystemObject;
512 
513  if (!handleSubsystemObjectCreation(pInstanceConfigurableElement, context,
514  bHasCreatedSubsystemObject, strError)) {
515 
516  return false;
517  }
518  // Check for creation error
519  if (bShouldCreateSubsystemObject && !bHasCreatedSubsystemObject) {
520 
521  strError = getMappingError("Not found",
522  "Subsystem object mapping key is missing",
523  pInstanceConfigurableElement);
524  return false;
525  }
526  // Not created and no error, keep diving
527  bKeepDiving = !bHasCreatedSubsystemObject;
528 
529  return true;
530 
531  default:
532  assert(0);
533  return false;
534  }
