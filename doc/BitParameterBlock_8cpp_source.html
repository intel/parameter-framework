62 void CBitParameterBlock::setDefaultValues(CParameterAccessContext &parameterAccessContext) const
63 {
64  // Default value is 0 as their is no min bound for bit parameters,
65  // thus 0 is always a valid value.
66  // BitParameterBlock can be as long a 64 bit, thus an 64 bit long variable
67  // is necessary to initialize it.
68  uint64_t uiDefaultValue = 0;
69 
70  // Write blackboard
71  CParameterBlackboard *pBlackboard = parameterAccessContext.getParameterBlackboard();
72 
73  // Beware this code works on little endian architectures only!
74  pBlackboard->writeInteger(&uiDefaultValue, getSize(),
75  getOffset() - parameterAccessContext.getBaseOffset());
76 }
