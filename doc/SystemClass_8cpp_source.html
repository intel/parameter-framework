107 bool CSystemClass::loadSubsystemsFromSharedLibraries(core::Results& errors,
108  const CSubsystemPlugins* pSubsystemPlugins)
109 {
110  // Plugin list
111  list<string> lstrPluginFiles;
112 
113  size_t pluginLocation;
114 
115  for (pluginLocation = 0; pluginLocation < pSubsystemPlugins->getNbChildren(); pluginLocation++) {
116 
117  // Get Folder for current Plugin Location
118  const CPluginLocation* pPluginLocation = static_cast<const CPluginLocation*>(pSubsystemPlugins->getChild(pluginLocation));
119 
120  string strFolder(pPluginLocation->getFolder());
121  if (!strFolder.empty()) {
122  strFolder += "/";
123  }
124  // Iterator on Plugin List:
125  list<string>::const_iterator it;
126 
127  const list<string>& pluginList = pPluginLocation->getPluginList();
128 
129  for (it = pluginList.begin(); it != pluginList.end(); ++it) {
130 
131  // Fill Plugin files list
132  lstrPluginFiles.push_back(strFolder + *it);
133  }
134  }
135 
136  // Actually load plugins
137  while (!lstrPluginFiles.empty()) {
138 
139  // Because plugins might depend on one another, loading will be done
140  // as an iteration process that finishes successfully when the remaining
141  // list of plugins to load gets empty or unsuccessfully if the loading
142  // process failed to load at least one of them
143 
144  // Attempt to load the complete list
145  if (!loadPlugins(lstrPluginFiles, errors)) {
146 
147  // Unable to load at least one plugin
148  break;
149  }
150  }
151 
152  if (!lstrPluginFiles.empty()) {
153  // Unable to load at least one plugin
154  errors.push_back("Unable to load the following plugins: " +
155  utility::asString(lstrPluginFiles, ", ") + ".");
156  return false;
157  }
158 
159  return true;
160 }

164 bool CSystemClass::loadPlugins(list<string>& lstrPluginFiles, core::Results& errors)
165 {
166  assert(lstrPluginFiles.size());
167 
168  bool bAtLeastOneSubsystemPluginSuccessfullyLoaded = false;
169 
170  list<string>::iterator it = lstrPluginFiles.begin();
171 
172  while (it != lstrPluginFiles.end()) {
173 
174  string strPluginFileName = *it;
175 
176  // Load attempt
177  try {
178  auto library = utility::make_unique<DynamicLibrary>(strPluginFileName);
179 
180  // Load symbol from library
181  auto subSystemBuilder =
182  library->getSymbol<PluginEntryPointV1>(entryPointSymbol);
183 
184  // Store libraries handles
185  _subsystemLibraryHandleList.push_back(std::move(library));
186 
187  // Fill library
188  subSystemBuilder(_pSubsystemLibrary, _logger);
189 
190  } catch (std::exception& e) {
191  errors.push_back(e.what());
192 
193  // Next plugin
194  ++it;
195 
196  continue;
197  }
198 
199  // Account for this success
200  bAtLeastOneSubsystemPluginSuccessfullyLoaded = true;
201 
202  // Remove successfully loaded plugin from list and select next
203  lstrPluginFiles.erase(it++);
204  }
205 
206  return bAtLeastOneSubsystemPluginSuccessfullyLoaded;
207 }
